export function Button({ name , url}) {
  const redirectToURL = () => {
    // Define the url of the location
    window.location.href = url;
  };
  //make a button with the name of the division that takes u to that page
  return (
    <div>
      <button className="rounded-button" onClick={redirectToURL}>{name}</button>
    </div>
  );
}
